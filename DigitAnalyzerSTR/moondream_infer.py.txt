#!/usr/bin/env python3
"""
moondream_infer.py — called by DigitAnalyzerSTR to analyze a single frame.

Usage:
    python moondream_infer.py <image_path> <mode> [columns_json]

    mode = "discover"  — find all labeled fields, return {label: value} JSON
    mode = "read"      — read values for known columns (pass columns as JSON array)

Outputs a single line of JSON to stdout.
Errors go to stderr only — stdout is always valid JSON.
"""

import sys
import json
import os

def preprocess_image(image):
    """Sharpen and boost contrast to improve digit recognition accuracy."""
    from PIL import ImageEnhance, ImageFilter
    image = image.convert("RGB")
    image = ImageEnhance.Contrast(image).enhance(2.0)
    image = ImageEnhance.Sharpness(image).enhance(3.0)
    image = image.filter(ImageFilter.SHARPEN)
    return image

def query_with_retry(model, image, prompt):
    """Run inference twice on uncertain results and return the better answer."""
    result1 = model.query(image, prompt)
    answer1 = result1.get("answer", "") if isinstance(result1, dict) else str(result1)

    # If result has question marks, try again and pick the better one
    if "?" in answer1:
        result2 = model.query(image, prompt)
        answer2 = result2.get("answer", "") if isinstance(result2, dict) else str(result2)
        return answer1 if answer1.count("?") <= answer2.count("?") else answer2

    return answer1

def extract_json(answer):
    """Extract and parse JSON object from model response."""
    start = answer.find("{")
    end   = answer.rfind("}") + 1
    if start >= 0 and end > start:
        try:
            return json.loads(answer[start:end])
        except json.JSONDecodeError:
            pass
    return None
